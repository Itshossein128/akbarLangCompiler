
sahih n;

// Get the input from the user
benvis("Enter a number to calculate factorial: ");
begir(n);

// Define variables for the calculation
sahih factorial = 1;
sahih i = 1;

// Calculate factorial using a loop
baraye(sahih i = 1; ta i <= n; i = i + 1) {
    factorial = factorial * i;
}

// Print the result
benvis("Factorial of ");
benvis(n);
benvis(" is: ");
benvis(factorial);
